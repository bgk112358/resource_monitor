平台SoC: Dual-Core Cortex-A53 @ 1.5GHz, RAM: ~512MB, OS: Linux
版本v1.0

2. CPU 测试

2.1 单核满载测试

# 绑定到 CPU0，持续 60 秒
taskset -c 0 stress-ng --cpu 1 --timeout 60s --metrics-brief

# 同时监控温度和频率
watch -n 1 'cat /sys/class/thermal/thermal_zone0/temp; cat /sys/devices/system/cpu/cpu0/cpufreq/scaling_cur_freq'

2.2 双核满载测试

# 双核 100% 占用
stress-ng --cpu 2 --timeout 120s --metrics-brief

# 实时监控 (另一终端)
htop -d 10

通过标准：双核满载 120s 内温度 ≤ 85°C，无降频，无 kernel panic。

2.3 上下文切换压力

# 大量线程上下文切换
stress-ng --cpu 2 --ctx 4 --timeout 60s --metrics-brief

# 查看切换率
perf stat -e context-switches stress-ng --cpu 2 --timeout 30s

通过标准：上下文切换率稳定，无 scheduler 告警。

2.4 CPU 占用率精确测量（单应用）

# 启动被测应用
./your_app &
PID=$!

# 采样 30 秒，每 1 秒一次
for i in $(seq 1 30); do
    ps -p $PID -o %cpu --no-headers >> cpu_sample.log
    sleep 1
done

# 计算平均 CPU 占用
awk '{sum+=$1} END {print "AVG CPU%:", sum/NR}' cpu_sample.log

3. 内存测试

3.1 常驻内存测量 (RSS/PSS/USS)

# PSS = 按共享库比例分摊后的真实内存占用 (最准确)
smem -P your_app -c "pid name pss rss uss"

# 持续监控
while true; do
    smem -P your_app -c "pid pss" --no-headers >> mem_pss.log
    sleep 2
done

3.2 内存泄漏检测

# Valgrind 检测 (仅开发环境，性能下降 10-20x)
valgrind --leak-check=full --log-file=vg.log ./your_app

# 生产环境：长时间运行监控
# 运行 24 小时，每 10 分钟采样一次内存
for i in $(seq 1 144); do
    smem -P your_app -c "pid pss" --no-headers >> leak_24h.log
    sleep 600
done
# 用 Excel 画 PSS 趋势线，斜率 ≈ 0 则无泄漏

3.3 OOM 阈值测试

# 逐步增加内存压力，观察 OOM Killer 触发点
stress-ng --vm 1 --vm-bytes 250M --timeout 30s  # 250MB
stress-ng --vm 1 --vm-bytes 280M --timeout 30s  # 280MB
stress-ng --vm 1 --vm-bytes 300M --timeout 30s  # 300MB (预期触发 OOM)

# 查看 OOM 日志
dmesg | grep -i "oom\|killed"

4. 存储测试

4.1 eMMC 读写性能

# 顺序写
dd if=/dev/zero of=/tmp/test.bin bs=1M count=100 oflag=direct 2>&1 | tail -1
# 顺序读
dd if=/tmp/test.bin of=/dev/null bs=1M count=100 iflag=direct 2>&1 | tail -1

# fio 综合测试
fio --name=emmc_test --filename=/tmp/fio.dat --size=200M \
    --rw=randrw --bs=4k --direct=1 --numjobs=1 --runtime=30 \
    --ioengine=libaio --group_reporting

4.2 NAND Flash 测试

# 查看 MTD 分区
cat /proc/mtd

# UBI 卷读写速度
dd if=/dev/zero of=/mnt/nand/test.bin bs=64k count=100 2>&1 | tail -1
dd if=/mnt/nand/test.bin of=/dev/null bs=64k count=100 2>&1 | tail -1

5. 网络测试

5.1 以太网带宽

# 服务端 (PC)
iperf3 -s

# 客户端 (TBox)
iperf3 -c 192.168.1.100 -t 30 -i 1     # TCP 上行
iperf3 -c 192.168.1.100 -t 30 -i 1 -R  # TCP 下行
iperf3 -c 192.168.1.100 -t 30 -u -b 100M  # UDP 测试

5.2 4G 网络质量

# 信号质量
qmicli -d /dev/cdc-wdm0 --nas-get-signal-strength

# 延迟和丢包
ping -c 100 -i 0.2 8.8.8.8 | tail -3

# 带宽 (需云端 iperf3 服务端)
iperf3 -c cloud-server.example.com -t 15 -i 1

5.3 网络重连测试

# 模拟断网 → 重连
ifconfig wwan0 down; sleep 30; ifconfig wwan0 up
# 记录从 down 到 ping 通的时间
timeout 60 sh -c 'while ! ping -c1 -W1 8.8.8.8 >/dev/null 2>&1; do sleep 1; done'

6. CANFD 总线测试

6.1 收发功能验证

# 设置 CANFD 接口 (2Mbps 数据段)
ip link set can0 type can bitrate 500000 dbitrate 2000000 fd on
ip link set can0 up

# 发送 CANFD 帧 (64字节数据)
cansend can0 123##1122334455667788AABBCCDDEEFF00112233445566778899AABBCCDDEEFF00112233445566778899AABBCCDDEEFF00

# 接收监控
candump can0 -x

6.2 高负载压力测试

# cangen 生成随机 CANFD 帧 (满载)
cangen can0 -g 0 -I 123 -L 64 -f

# 同时监控错误
ip -details -statistics link show can0 | grep -E "errors|dropped"

通过标准：1Mbps 仲裁段 / 2Mbps 数据段满载运行 60s，rx_dropped = 0，errors = 0。

7. SPI 通信测试

# 标准 spidev 测试 (发 32 字节 → 收 32 字节)
spidev_test -D /dev/spidev0.0 -s 10000000 -b 8 -v

# 自定义 DMA 回环测试
./spi_dma_loopback -d /dev/spidev0.0 -s 10000000 -l 4096 -c 1000
# 参数: 10MHz, 4KB 块, 1000 次 → 4MB 总传输量, 统计 CRC 错误

通过标准：10MHz 时钟，1000 次 4KB 传输，CRC 错误 = 0。

8. RS485 通信测试

# 设置 RS485 模式
echo 1 > /sys/class/tty/ttyRS0/rs485

# 回环测试 (短接 A-B 线)
cat /dev/ttyRS0 > recv.log &  # 后台接收
echo "TEST_PATTERN_0123456789ABCDEF" > /dev/ttyRS0
sleep 1; kill %1
grep "TEST_PATTERN" recv.log && echo "✅ RS485 OK" || echo "❌ RS485 FAIL"

# Modbus 通信验证
modbus-rtu-test -d /dev/ttyRS0 -b 115200 -a 1 -r 0x0000 -c 10

9. GPIO / ADC / PWM 测试

9.1 GPIO 中断响应

# 监听 GPIO 中断
gpiomon -n 1 -r gpiochip0 17   # GPIO17 上升沿, 单次触发

# 用示波器量: 物理信号跳变 → gpiomon 打印时间差

9.2 ADC 采样

# 读取 ADC 通道
cat /sys/bus/iio/devices/iio:device0/in_voltage0_raw
# 换算: 实际电压 = raw * Vref / 4096 (12-bit ADC)

# 采样 100 次统计精度
for i in $(seq 1 100); do cat /sys/bus/iio/devices/iio:device0/in_voltage0_raw >> adc.log; sleep 0.01; done
# 计算标准差
awk '{sum+=$1; sumsq+=$1*$1} END {printf "AVG=%.1f STD=%.2f\n", sum/NR, sqrt(sumsq/NR - (sum/NR)^2)}' adc.log

9.3 PWM 输出

# 设置 PWM 占空比
echo 0 > /sys/class/pwm/pwmchip0/export
echo 20000 > /sys/class/pwm/pwmchip0/pwm0/period     # 20kHz
echo 10000 > /sys/class/pwm/pwmchip0/pwm0/duty_cycle  # 50%
echo 1 > /sys/class/pwm/pwmchip0/pwm0/enable

# 用示波器验证实际占空比和频率

10. GPS / PPS 时间同步测试

# GPS 定位状态
gpspipe -w -n 20 | grep -E "TPV|mode"

# PPS 脉冲精度
ppstest /dev/pps0
# 预期: offset < 1000ns, jitter < 1000ns

# NTP 同步偏差
ntpq -c rv | grep -E "offset|jitter"
# 预期: offset < 10ms, jitter < 1ms

# 长时间 PPS 稳定性 (1小时)
timeout 3600 ppstest /dev/pps0 | awk '{print $3}' > pps_jitter_1h.log

通过标准：PPS offset < 1μs，NTP offset < 10ms，GPS 定位 3D fix 时间 < 60s（冷启动）。

11. HSM 安全芯片测试

# PKCS#11 密钥列表
pkcs11-tool --module /usr/lib/libpkcs11.so -L

# ECDSA 签名性能 (Key-0)
pkcs11-tool --module /usr/lib/libpkcs11.so -s -l -d 0 --id 0 \
    -i /tmp/hash.bin -o /tmp/sig.bin
# 循环 100 次计时
time for i in $(seq 1 100); do
    pkcs11-tool --module /usr/lib/libpkcs11.so -s -l --id 0 \
        -i /tmp/hash.bin -o /tmp/sig.bin 2>/dev/null
done
# OpenSSL 引擎验证 (TLS 握手测试)
openssl speed -engine pkcs11 -elapsed -evp ecdsa 2>&1 | grep -E "sign|verify"

# 随机数生成质量
pkcs11-tool --module /usr/lib/libpkcs11.so --generate-random 1024 \
    | xxd | head

通过标准：ECDSA secp256r1 sign ≥ 50 次/s，随机数熵值正常。

12. 应用程序资源占用测试 （每个应用通用）

以下方法适用于 TBox 平台上的任何应用程序。将被测应用的进程名替换 YOUR_APP 即可。

12.1 一键采集脚本

#!/bin/sh
# app_resource_profile.sh — 应用资源画像采集
# 用法: ./app_resource_profile.sh YOUR_APP 30

APP_NAME="${1:?Usage: $0  }"
DURATION="${2:-30}"
OUTDIR="/tmp/resource_profile_${APP_NAME}_$(date +%Y%m%d_%H%M%S)"
mkdir -p "$OUTDIR"

PID=$(pidof "$APP_NAME" 2>/dev/null | awk '{print $1}')
if [ -z "$PID" ]; then
    echo "❌ 进程 '$APP_NAME' 未运行"
    exit 1
fi
echo "📊 采集进程 $APP_NAME (PID=$PID) 的资源数据, 持续 ${DURATION}s..."

# ▼ CPU 每秒采样
echo "sec,cpu_percent" > "$OUTDIR/cpu.csv"
for i in $(seq 1 $DURATION); do
    cpu=$(ps -p $PID -o %cpu --no-headers 2>/dev/null | tr -d ' ')
    echo "$i,${cpu:-0}" >> "$OUTDIR/cpu.csv"
    sleep 1
done

# ▼ 内存 PSS (采样 5 次)
echo "sample,pss_kb,rss_kb,uss_kb" > "$OUTDIR/mem.csv"
for i in 1 2 3 4 5; do
    smem -P "$APP_NAME" -c "pid pss rss uss" --no-headers 2>/dev/null | head -1 | \
        awk -v i="$i" '{printf "%d,%s,%s,%s\n", i, $2, $3, $4}' >> "$OUTDIR/mem.csv"
    sleep $((DURATION / 5))
done

# ▼ 线程数 + FD 数
echo "sample,threads,fd_count" > "$OUTDIR/threads_fd.csv"
for i in 1 2 3 4 5; do
    t_cnt=$(ls /proc/$PID/task 2>/dev/null | wc -l)
    fd_cnt=$(ls /proc/$PID/fd 2>/dev/null | wc -l)
    echo "$i,${t_cnt},${fd_cnt}" >> "$OUTDIR/threads_fd.csv"
    sleep $((DURATION / 5))
done

# ▼ IO 读写量 (需要 root 或 CAP_SYS_PTRACE)
echo "sample,read_kb,write_kb" > "$OUTDIR/io.csv"
for i in 1 2 3 4 5; do
    if [ -r /proc/$PID/io ]; then
        rchar=$(awk '/read_bytes/ {print $2}' /proc/$PID/io)
        wchar=$(awk '/write_bytes/ {print $2}' /proc/$PID/io)
        echo "$i,$((rchar/1024)),$((wchar/1024))" >> "$OUTDIR/io.csv"
    fi
    sleep $((DURATION / 5))
done

# ▼ 网络连接数
ss -tnp 2>/dev/null | grep "pid=$PID" | wc -l > "$OUTDIR/net_connections.txt"

# ▼ 汇总报告
cat > "$OUTDIR/report.txt" << EOF
========================================
 应用资源画像报告
========================================
进程名称:  $APP_NAME
PID:       $PID
采集时长:  ${DURATION}s
采集时间:  $(date)

CPU 平均占用:  $(awk -F',' 'NR>1 {sum+=$2;n++} END {printf "%.1f%%", sum/n}' "$OUTDIR/cpu.csv")
CPU 峰值占用:  $(awk -F',' 'NR>1 {if($2>max) max=$2} END {printf "%.1f%%", max}' "$OUTDIR/cpu.csv")
内存 PSS:      $(awk -F',' 'NR>1 {sum+=$2;n++} END {printf "%.0f KB (%.1f MB)", sum/n, sum/n/1024}' "$OUTDIR/mem.csv")
内存 RSS:      $(awk -F',' 'NR>1 {sum+=$3;n++} END {printf "%.0f KB (%.1f MB)", sum/n, sum/n/1024}' "$OUTDIR/mem.csv")
线程数:        $(awk -F',' 'NR>1 {sum+=$2;n++} END {printf "%.0f", sum/n}' "$OUTDIR/threads_fd.csv")
FD 数:         $(awk -F',' 'NR>1 {sum+=$3;n++} END {printf "%.0f", sum/n}' "$OUTDIR/threads_fd.csv")
网络连接数:    $(cat "$OUTDIR/net_connections.txt")

详细数据:
  CPU 采样:    $OUTDIR/cpu.csv
  内存采样:    $OUTDIR/mem.csv
  IO 采样:     $OUTDIR/io.csv
  线程/FD:     $OUTDIR/threads_fd.csv
========================================
EOF

cat "$OUTDIR/report.txt"
echo ""
echo "✅ 所有数据保存在: $OUTDIR/"

12.2 手动单项测量

# ── CPU: 平均 + 峰值 ──
PID=$(pidof YOUR_APP | awk '{print $1}')
for i in $(seq 1 30); do
    ps -p $PID -o %cpu --no-headers | tr -d ' '
    sleep 1
done | awk '{sum+=$1; n++; if($1>max) max=$1}
            END {printf "CPU AVG=%.1f%%  PEAK=%.1f%%\n", sum/n, max}'

# ── 内存: PSS (最准) / RSS / USS ──
smem -P YOUR_APP -c "pid name pss rss uss"

# ── 线程数 ──
ls /proc/$PID/task | wc -l

# ── 打开的文件描述符 ──
ls /proc/$PID/fd | wc -l
# 哪些文件被打开?
ls -l /proc/$PID/fd | awk '{print $NF}' | sort | uniq -c | sort -rn

# ── 磁盘 IO (累计) ──
cat /proc/$PID/io
# read_bytes:  累计读 (字节)
# write_bytes: 累计写 (字节)
# 隔 10 秒再读一次, 差值 / 10 = 平均 IO 吞吐 (B/s)

# ── 网络流量 (累计) ──
cat /proc/$PID/net/dev
# 或使用 nethogs (实时)
nethogs -d 2 $PID

12.3 长时间稳定性测试（24 小时）

#!/bin/sh
# 24h 稳定性监控
APP_NAME="$1"
PID=$(pidof "$APP_NAME" | awk '{print $1}')
LOG="/tmp/stability_${APP_NAME}.csv"
echo "timestamp,cpu%,pss_kb,threads,fd" > "$LOG"

for i in $(seq 1 8640); do   # 8640 × 10s = 24h
    cpu=$(ps -p $PID -o %cpu --no-headers 2>/dev/null | tr -d ' ')
    pss=$(smem -P "$APP_NAME" -c "pss" --no-headers 2>/dev/null | head -1 | tr -d ' ')
    threads=$(ls /proc/$PID/task 2>/dev/null | wc -l)
    fd=$(ls /proc/$PID/fd 2>/dev/null | wc -l)
    echo "$(date +%H:%M:%S),${cpu:-0},${pss:-0},${threads:-0},${fd:-0}" >> "$LOG"
    sleep 10
done

# 分析: 用 Excel 打开 LOG, 画 PSS 趋势线
# 判断: 斜率 ≈ 0 → 无泄漏; 斜率 > 0 → 有泄漏, 斜率越大越严重

14. 综合压力测试（全系统）

同时加载所有资源，模拟最坏场景：

#!/bin/sh
# TBox 综合压力测试脚本 (运行 10 分钟)

# 1. CPU: 双核 80% 占用
stress-ng --cpu 2 --cpu-load 80 --timeout 600s &

# 2. 内存: 分配 200MB
stress-ng --vm 1 --vm-bytes 200M --timeout 600s &

# 3. 存储: 持续随机读写
fio --name=stress_emmc --filename=/tmp/stress.dat --size=50M \
    --rw=randrw --bs=4k --direct=1 --numjobs=1 --runtime=600 \
    --ioengine=libaio --group_reporting &

# 4. CANFD: 满载发送
cangen can0 -g 0 -L 64 -f &
cangen can1 -g 0 -L 64 -f &

# 5. 网络: iperf3 持续测试 (需服务端)
iperf3 -c 192.168.1.100 -t 600 -i 10 &

# 6. GPS: 持续读取
while true; do gpspipe -w -n 1 >> /dev/null; sleep 0.1; done &

# 7. 等待结束
wait
echo "=== 综合压力测试完成 ==="
dmesg | tail -50 | grep -iE "error|fail|oom|panic"

通过标准：10 分钟全负载运行，无 OOM Kill、无 kernel panic、无 CAN 丢帧、网络丢包 < 0.5%。
